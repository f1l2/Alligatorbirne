list-dev

list-ep

list-ds --devId 1

list-dev-by --devInfo Temperture --domInfo FLOOR1

rq --epId 1 --query "CONDITION name = 'Temperature' AND value >= 25" --queryName detectTemperatureAbove25

rq --epId 1 --query "CONDITION name = 'Temperature' AND value <= 15" --queryName detectTemperatureUnder25

rr --epId 1 --rule 'detectTemperatureAbove25 TRIGGERS Temperature, FLOOR1, TASK_INTERVAL_MS = 2000' --ruleName decreaseIntervalBasedOnHighTemperature

rr --epId 1 --rule 'detectTemperatureUnder25 TRIGGERS Temperature, FLOOR1, TASK_INTERVAL_MS = 10000' --ruleName increaseIntervalBasedOnLowTemperature

activate-rule --epId 1 --ruleName decreaseIntervalBasedOnHighTemperature

activate-rule --epId 1 --ruleName increaseIntervalBasedOnLowTemperature
